#include <iostream>
#include <cmath>

class Complex {
private:
    double re; // real part
    double im; // imaginary part

public:
    // Default constructor initializes re and im to 0
    Complex() : re(0), im(0) {}
    // Custom constructor that takes only the real part
    Complex(double real) : re(real), im(0) {}
    // Custom constructor that takes both real and imaginary parts
    Complex(double real, double imag) : re(real), im(imag) {}
    // Function to calculate the magnitude of the complex number
    double calculateMagnitude() const {
        return std::sqrt(re * re + im * im);
    }

    // Function to add two complex numbers
    Complex add(const Complex& other) const {
        return Complex(re + other.re, im + other.im);
    }

    // Function to subtract two complex numbers
    Complex subtract(const Complex& other) const {
        return Complex(re - other.re, im - other.im);
    }
    // Function to multiply two complex numbers
    Complex multiply(const Complex& other) const {
        double realPart = re * other.re - im * other.im;
        double imagPart = re * other.im + im * other.re;
        return Complex(realPart, imagPart);
    }
    // Function to divide two complex numbers
    Complex divide(const Complex& other) const {
        double denominator = other.re * other.re + other.im * other.im;
        double realPart = (re * other.re + im * other.im) / denominator;
        double imagPart = (im * other.re - re * other.im) / denominator;
        return Complex(realPart, imagPart);
    }

    // Function to display the complex number
    void display() const {
        std::cout << re << " + " << im << "j" << std::endl;
    }

};


int main() {
    double real1, imag1, real2, imag2;
    std::cout << "Enter the real and imaginary parts of the first complex number (a + jb) as 'a b': ";
    std::cin >> real1 >> imag1;
    Complex c1(real1, imag1);

    std::cout << "Enter the real and imaginary parts of the second complex number (a + jb) as 'a b': ";
    std::cin >> real2 >> imag2;
    Complex c2(real2, imag2);

    std::cout << "First complex number: ";
    c1.display();
    std::cout << "Second complex number: ";
    c2.display();

    Complex sum = c1.add(c2);
    Complex difference = c1.subtract(c2);
    Complex product = c1.multiply(c2);
    Complex quotient = c1.divide(c2);

    std::cout << "Sum: ";
    sum.display();
    std::cout << "Difference: ";
    difference.display();
    std::cout << "Product: ";
    product.display();
    std::cout << "Quotient: ";
    quotient.display();

    return 0;
}
