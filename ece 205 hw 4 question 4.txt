#include <iostream>
using namespace std;

void displayBoard(char board[]) { // displays the current state of the tic-tac-toe board
    for (int i = 0; i < 9; i++) { // loop through each cell of the board and display its value (either a number for an empty cell or 'X'/'O' for occupied cells)
        cout << board[i] << " ";
        if ((i + 1) % 3 == 0) cout << endl; // after every 3 cells, print a new line to format the board as a 3x3 grid
    }
}

bool checkWin(char b[], char p) { // checks if the player 'p' has won by checking all possible winning combinations (rows, columns, diagonals)
    return (b[0]==p && b[1]==p && b[2]==p) || 
           (b[3]==p && b[4]==p && b[5]==p) ||
           (b[6]==p && b[7]==p && b[8]==p) ||
           (b[0]==p && b[3]==p && b[6]==p) ||
           (b[1]==p && b[4]==p && b[7]==p) ||
           (b[2]==p && b[5]==p && b[8]==p) ||
           (b[0]==p && b[4]==p && b[8]==p) ||
           (b[2]==p && b[4]==p && b[6]==p);
}

int main() {
    char board[9] = {'1','2','3','4','5','6','7','8','9'}; // initialize the tic-tac-toe board with numbers representing empty cells
    char player = 'X'; // variable to keep track of the current player, starting with 'X'
    int move; // variable to hold the player's move input

    for (int i = 0; i < 9; i++) { // loop to allow up to 9 moves (the maximum number of moves in a tic-tac-toe game) and check for a win after each move
        displayBoard(board); // display the current state of the board to the user before each move

        cout << "Player " << player << ", enter position: ";
        cin >> move;

        if (move < 1 || move > 9 || board[move - 1] == 'X' || board[move - 1] == 'O') { // validate the player's move to ensure it's a number between 1 and 9 and that the chosen cell is not already occupied by 'X' or 'O'
            cout << "Invalid move!\n";
            i--; // decrement the move counter to allow the player to try again without losing their turn
            continue;
        }

        board[move - 1] = player; // update the board with the player's move by placing 'X' or 'O' in the chosen cell

        if (checkWin(board, player)) { // check if the current player has won after making their move by calling the checkWin function
            displayBoard(board); // display the final state of the board to the user after a win is detected
            cout << "Player " << player << " wins!\n";
            return 0;
        }

        player = (player == 'X') ? 'O' : 'X'; // switch the player for the next turn (if the current player is 'X', switch to 'O', and vice versa)
    }

    cout << "It's a draw!\n";

    return 0;
}
