#include <iostream>
#include <fstream>
#include <cmath>
#include <iomanip>
using namespace std;

//Question 2
double average (double scores[], int size)
{
    double sum = 0.0;
    for (int i = 0; i < size; i++)
    {
        sum += scores[i];
    }
    return sum / size;
}

double st_dev(double scores[], int size, double avg)
{
    double sum_squared_diff = 0.0;
    for (int i = 0; i < size; i++)
    {
        sum_squared_diff += pow(scores[i] - avg, 2);
    }
    return sqrt(sum_squared_diff / size);
}
int main()
{
    const int size = 5; //maximum number of scores
    double scores[size]; //array to hold scores
    
    ifstream input_file("scores.txt"); //input file stream
    ofstream output_file("output.txt"); //output file stream

    for (int i = 0; i < size; i++)
    {
        input_file >> scores[i]; //read scores from file
    }

    double avg = average(scores, size); //calculate average
    double std_dev = st_dev(scores, size, avg); //calculate standard deviation

    cout << fixed << setprecision(2); //output to 2 decimal places
    output_file << fixed << setprecision(2); //output to 2 decimal places in file

    //output results
    cout << "Average: " << avg << endl;
    cout << "Standard Deviation: " << std_dev << endl;

    output_file << "Average: " << avg << endl;
    output_file << "Standard Deviation: " << std_dev << endl;

    input_file.close(); //close input file
    output_file.close(); //close output file

    return 0;
}
