//Purpose of program: To build a wearable fall detection and emergency alert system that detects falls using a tilt sensor, confirms with an IR sensor, counts down before triggering an alert, and allows cancellation. The alert should include a UART message, servo flag raise, and an LED indicator.
//Program Inputs: Tilt sensor on PE2, IR sensor on PE4, Cancel button on PF0, SW1 on PF4, Potentiometer on PD0
//Program Outputs: Green LED on PF1, Yellow LED on PF2, Red LED on PF3, UART on PA0/PA1, Servo motor on PB7
//*************************************************


//File Include Statements
#include <stdbool.h>
#include <stdint.h>
#include "inc/tm4c123gh6pm.h"
#include "inc/hw_memmap.h"
#include "inc/hw_types.h"
#include "driverlib/sysctl.h"
#include "driverlib/interrupt.h"
#include "driverlib/gpio.h"
#include "driverlib/systick.h"
#include "driverlib/uart.h"
#include "driverlib/pin_map.h"
#include "driverlib/adc.h"
#include "driverlib/pwm.h"


//variable declarations
#define MONITORING    0
#define FALL_DETECTED 1
#define COUNTDOWN     2
#define ALERT         3
#define RESET         4

#define GPIO_PA0_U0RX 0x00000001 //UART receiver
#define GPIO_PA1_U0TX 0x00000401 //UART transmitter

#define SERVO_DOWN_US  1000  //1ms pulse - down position
#define SERVO_UP_US    2000  //2ms pulse - up position

struct state {
   int greenLED;   //1 = green LED on in this state, 0=off
   int yellowLED;  //1 = yellow LED on in this state
   int redLED;     //1 = red LED on in this state
};

typedef struct state State_t;

//FSM table
State_t fsm[5] = {
//   green  yellow  red
   { 1,     0,      0 },   //MONITORING:    green on
   { 0,     0,      0 },   //FALL_DETECTED: all off, waiting for IR confirmation
   { 0,     1,      0 },   //COUNTDOWN:     yellow on
   { 0,     0,      1 },   //ALERT:         red on
   { 0,     0,      0 },   //RESET:         all off
};

int currentState   = MONITORING; //keeps track of current state. system starts in monitoring when powered on
int countdownTicks = 0;           //tracks seconds elapsed in COUNTDOWN and FALL_DETECTED
int countdownLimit = 10;          //how many seconds before alert triggers, set by potentiometer
unsigned long ulPeriod;           //stores PWM period in clock ticks


//function prototypes
void portF_setup(void);
void portF_cancel_handler(void);
void portE_input_setup(int input_pin);
void portE_int_handler(void);
void SysTick_Handler(void);
void updateLEDs(void);
void uart_setup(void);
void sendAlert(void);
void adc_setup(void);
int  readPotentiometer(void);
void servo_setup(void);
void servoSetPulse(uint32_t pulse_us);
void servoUp(void);
void servoDown(void);
void watchdog_setup(void);


int main(void) {

   SysCtlClockSet(SYSCTL_SYSDIV_4 | SYSCTL_USE_PLL | SYSCTL_XTAL_16MHZ | SYSCTL_OSC_MAIN); //set the clock

   portF_setup();           //set up port F for LEDs and buttons
   portE_input_setup(0x14); //PE2 for tilt sensor, PE4 for IR sensor
   uart_setup();            //set up UART on PA0 and PA1
   adc_setup();             //set up ADC on PD0 for potentiometer
   servo_setup();           //set up PWM on PB7 for servo motor
   watchdog_setup();        //set up watchdog timer

   SysTickPeriodSet(50000000);          //fires every 1 second (50MHz)
   SysTickIntRegister(SysTick_Handler); //connect systick to handler
   SysTickIntEnable();                  //turn on systick interrupt
   SysTickEnable();                     //start systick

   IntMasterEnable(); //enable global interrupts

   GPIO_PORTF_DATA_R &= ~0x0E; //start with all LEDs off

   int adcValue = readPotentiometer(); //read potentiometer once at startup
   countdownLimit = (adcValue / 4095.0) * 25 + 5; //maps potentiometer to 5-30 seconds

   servoDown(); //make sure flag starts in down position
   SysCtlDelay(SysCtlClockGet() / 2); //wait for servo to reach down position

   updateLEDs(); //set LEDs to match starting state

   while(1) {

       // refresh watchdog only while in monitoring so it resets if system gets stuck
       if (currentState == MONITORING) {
           WATCHDOG0_LOAD_R = SysCtlClockGet() * 2; //refresh with 2 second timeout
       }

       // SW1 on PF4 cycles ALERT and RESET for testing
       if (0 == (GPIO_PORTF_DATA_R & 0x10)) {
           SysCtlDelay(SysCtlClockGet() / 100); //debounce
           if (currentState == ALERT || currentState == RESET) {
               currentState = (currentState + 1) % 5;
               if (currentState == MONITORING) {
                   servoDown();
                   SysCtlDelay(SysCtlClockGet() / 2); //wait for servo to reach down
               }
               updateLEDs();
           }
           while (0 == (GPIO_PORTF_DATA_R & 0x10)) {} //wait for button release
           SysCtlDelay(SysCtlClockGet() / 10); //short pause after release
       }
   }

   return 0;
}


//cancels ALERT and returns to MONITORING when cancel button on PF0 is pressed during COUNTDOWN
void portF_cancel_handler(void) {

   if (currentState == COUNTDOWN) {
       countdownTicks = 0;
       currentState = MONITORING;
       updateLEDs();
   }

   GPIOIntClear(GPIO_PORTF_BASE, GPIO_PIN_0); //clear flag so handler can fire again
}


//handles timeout in FALL DETECTED and COUNTDOWN in ALERT. 
//runs every 1 second from systick
void SysTick_Handler(void) {

   if (currentState == FALL_DETECTED) {
       countdownTicks++;

//if IR sensor does not confirm within 10 seconds, assume false tilt and go back to MONITORING
       if (countdownTicks >= 10) {
           countdownTicks = 0;
           currentState = MONITORING;
           updateLEDs();
       }
   }

   else if (currentState == COUNTDOWN) {
       countdownTicks++;

       if (countdownTicks >= countdownLimit) { //countdown expired, trigger alert
           countdownTicks = 0;
           currentState = ALERT;
           updateLEDs();
           sendAlert();  //send emergency message over UART
           servoUp();    //raise the flag
       }
   }

   else {
       countdownTicks = 0; //reset counter in any other state
   }
}


//runs when tilt sensor or IR sensor changes
void portE_int_handler(void) {

   if (currentState == MONITORING) {
       if (0 == (GPIO_PORTE_DATA_R & 0x04)) { //PE2 low means fall detected
           countdownTicks = 0;
           currentState = FALL_DETECTED;
           updateLEDs();
       }
   }

   if (currentState == FALL_DETECTED) {
       if (0 == (GPIO_PORTE_DATA_R & 0x10)) { //PE4 low means person detected
           countdownTicks = 0;
           currentState = COUNTDOWN;
           updateLEDs();
       }
   }

   GPIOIntClear(GPIO_PORTE_BASE, GPIO_PIN_2 | GPIO_PIN_4); //clear flags so handler can fire again
}


//sets LEDs based on current state
void updateLEDs(void) {
   GPIO_PORTF_DATA_R &= ~0x0E; //clear all LEDs first

   if (fsm[currentState].greenLED)  GPIO_PORTF_DATA_R |= 0x02; //green on PF1
   if (fsm[currentState].yellowLED) GPIO_PORTF_DATA_R |= 0x04; //yellow on PF2
   if (fsm[currentState].redLED)    GPIO_PORTF_DATA_R |= 0x08; //red on PF3
}


//sends emergency alert message over UART one character at a time
void sendAlert(void) {
   UARTCharPut(UART0_BASE, 'F');
   UARTCharPut(UART0_BASE, 'A');
   UARTCharPut(UART0_BASE, 'L');
   UARTCharPut(UART0_BASE, 'L');
   UARTCharPut(UART0_BASE, ' ');
   UARTCharPut(UART0_BASE, 'D');
   UARTCharPut(UART0_BASE, 'E');
   UARTCharPut(UART0_BASE, 'T');
   UARTCharPut(UART0_BASE, 'E');
   UARTCharPut(UART0_BASE, 'C');
   UARTCharPut(UART0_BASE, 'T');
   UARTCharPut(UART0_BASE, 'E');
   UARTCharPut(UART0_BASE, 'D');
   UARTCharPut(UART0_BASE, ' ');
   UARTCharPut(UART0_BASE, '-');
   UARTCharPut(UART0_BASE, ' ');
   UARTCharPut(UART0_BASE, 'C');
   UARTCharPut(UART0_BASE, 'A');
   UARTCharPut(UART0_BASE, 'L');
   UARTCharPut(UART0_BASE, 'L');
   UARTCharPut(UART0_BASE, ' ');
   UARTCharPut(UART0_BASE, 'F');
   UARTCharPut(UART0_BASE, 'O');
   UARTCharPut(UART0_BASE, 'R');
   UARTCharPut(UART0_BASE, ' ');
   UARTCharPut(UART0_BASE, 'H');
   UARTCharPut(UART0_BASE, 'E');
   UARTCharPut(UART0_BASE, 'L');
   UARTCharPut(UART0_BASE, 'P');
   UARTCharPut(UART0_BASE, '!');
   UARTCharPut(UART0_BASE, '\n');
   UARTCharPut(UART0_BASE, '\r');
}


//converts a pulse width in microseconds to PWM ticks and sets servo position
void servoSetPulse(uint32_t pulse_us) {
   uint32_t pwmClock = SysCtlClockGet() / 64;
   uint32_t pulseTicks = (pwmClock * pulse_us) / 1000000;
   PWMPulseWidthSet(PWM0_BASE, PWM_OUT_1, pulseTicks);
}


//raises the servo flag to alert position
void servoUp(void) {
   servoSetPulse(SERVO_UP_US);
}


//returns the servo flag to down position
void servoDown(void) {
   servoSetPulse(SERVO_DOWN_US);
}


//sets up port F - LEDs as outputs, PF4 as SW1, PF0 as cancel button
void portF_setup(void) {
   SysCtlPeripheralEnable(SYSCTL_PERIPH_GPIOF); //clock for port F
   while (!SysCtlPeripheralReady(SYSCTL_PERIPH_GPIOF));
   GPIO_PORTF_LOCK_R  = 0x4C4F434B; //unlock port F
   GPIO_PORTF_CR_R   |= 0x11;
   GPIO_PORTF_DIR_R  |= 0x0E;   //PF1, PF2, PF3 as outputs for LEDs
   GPIO_PORTF_DIR_R  &= ~0x11;  //PF0 and PF4 as inputs for button inputs
   GPIO_PORTF_PUR_R  |= 0x11;   //pull up resistors for buttons
   GPIO_PORTF_DEN_R  |= 0x1F;   //digital enable on all 5 pins

   GPIOIntTypeSet(GPIO_PORTF_BASE, GPIO_PIN_0, GPIO_FALLING_EDGE); //trigger on press
   GPIOIntRegister(GPIO_PORTF_BASE, portF_cancel_handler); //register handler
   GPIOIntEnable(GPIO_PORTF_BASE, GPIO_PIN_0); //enable interrupt
}


//sets up PE2 and PE4 as inputs for tilt sensor and IR sensor
void portE_input_setup(int input_pin) {
   SysCtlPeripheralEnable(SYSCTL_PERIPH_GPIOE); //clock for port E
   while (!SysCtlPeripheralReady(SYSCTL_PERIPH_GPIOE));
   GPIO_PORTE_DIR_R  &= ~input_pin; //set as inputs
   GPIO_PORTE_PUR_R  |=  input_pin; //pull up resistor so pins read high when nothing detected
   GPIO_PORTE_DEN_R  |=  input_pin; //digital enable

   GPIOIntTypeSet(GPIO_PORTE_BASE, GPIO_PIN_2 | GPIO_PIN_4, GPIO_FALLING_EDGE);
   GPIOIntRegister(GPIO_PORTE_BASE, portE_int_handler); //register handler
   GPIOIntEnable(GPIO_PORTE_BASE, GPIO_PIN_2 | GPIO_PIN_4); //enable interrupts
}


//sets up UART0 on PA0 and PA1 for serial communication
void uart_setup(void) {
   SysCtlPeripheralEnable(SYSCTL_PERIPH_UART0); //enable UART hardware
   SysCtlPeripheralEnable(SYSCTL_PERIPH_GPIOA); //enable port A
   GPIOPinConfigure(GPIO_PA0_U0RX); //configure PA0 for UART RX
   GPIOPinConfigure(GPIO_PA1_U0TX); //configure PA1 for UART TX
   GPIOPinTypeUART(GPIO_PORTA_BASE, GPIO_PIN_0 | GPIO_PIN_1); //set pins for UART
   UARTConfigSetExpClk(UART0_BASE, SysCtlClockGet(), 115200,
   (UART_CONFIG_WLEN_8 | UART_CONFIG_STOP_ONE | UART_CONFIG_PAR_NONE));
}


//sets up ADC0 to read potentiometer on PD0
void adc_setup(void) {
   SysCtlPeripheralEnable(SYSCTL_PERIPH_ADC0); //enable ADC hardware
   ADCSequenceConfigure(ADC0_BASE, 0, ADC_TRIGGER_PROCESSOR, 0);
   ADCSequenceStepConfigure(ADC0_BASE, 0, 0, ADC_CTL_IE | ADC_CTL_END | ADC_CTL_CH7); //channel 7 = PD0
   ADCSequenceEnable(ADC0_BASE, 0);
   ADCIntClear(ADC0_BASE, 0); //clear any pending interrupt flag
}


//reads potentiometer value from ADC and returns a number between 0 and 4095
int readPotentiometer(void) {
   unsigned long adcValue;
   ADCProcessorTrigger(ADC0_BASE, 0); //trigger ADC to take a reading
   SysCtlDelay(SysCtlClockGet() / 3000); //wait for conversion to finish
   ADCSequenceDataGet(ADC0_BASE, 0, &adcValue); //get the reading
   return adcValue;
}


//sets up PWM on PB7 for servo motor
void servo_setup(void) {
   SysCtlPWMClockSet(SYSCTL_PWMDIV_64); //divide PWM clock by 64

   SysCtlPeripheralEnable(SYSCTL_PERIPH_PWM0);   //enable PWM module 0
   SysCtlPeripheralEnable(SYSCTL_PERIPH_GPIOB);  //enable port B
   while (!SysCtlPeripheralReady(SYSCTL_PERIPH_PWM0));
   while (!SysCtlPeripheralReady(SYSCTL_PERIPH_GPIOB));

   GPIO_PORTB_AFSEL_R |= 0x80; //enable alternate function on PB7
   GPIO_PORTB_PCTL_R  &= ~0xF0000000; //clear PB7 function bits
   GPIO_PORTB_PCTL_R  |=  0x40000000; //set PB7 to M0PWM1
   GPIO_PORTB_AMSEL_R &= ~0x80; //disable analog on PB7
   GPIO_PORTB_DEN_R   |=  0x80; //digital enable PB7
   GPIOPinTypePWM(GPIO_PORTB_BASE, GPIO_PIN_7); //set PB7 as PWM pin

   ulPeriod = (SysCtlClockGet() / 64) / 50; //15625 ticks for 50 Hz

   PWMGenConfigure(PWM0_BASE, PWM_GEN_0, PWM_GEN_MODE_DOWN | PWM_GEN_MODE_NO_SYNC);
   PWMGenPeriodSet(PWM0_BASE, PWM_GEN_0, ulPeriod);  //set 50Hz period
   servoDown(); //start in down position
   PWMGenEnable(PWM0_BASE, PWM_GEN_0); //enable PWM generator
   PWMOutputState(PWM0_BASE, PWM_OUT_1_BIT, true); //enable PWM output on PB7
}


//sets up watchdog timer and resets the system if it gets stuck
void watchdog_setup(void) {
   SYSCTL_RCGCWD_R |= 0x01; //enable watchdog clock
   while ((SYSCTL_PRWD_R & 0x01) == 0) {} //wait for it to be ready
   WATCHDOG0_LOAD_R = SysCtlClockGet() * 2; //set 2 second timeout
   WATCHDOG0_CTL_R |= 0x01; //enable watchdog with reset on timeout
}
